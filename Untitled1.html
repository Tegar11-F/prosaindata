import scrapy
from scrapy.crawler import CrawlerProcess
import pandas as pd

class MySpider(scrapy.Spider): name = 'my_spider' 
start_urls = ['https://play.google.com/store/apps/details?id=com.lazada.android&hl=id&gl=US']
def parse(self, response):
    # Contoh: Mendapatkan judul halaman web
    title = response.css('title::text').get()
    print("Judul halaman web:", title)
    # Contoh: Mendapatkan semua tautan pada halaman web
    links = response.css('a::attr(href)').getall()
    print("Tautan pada halaman web:")
    for link in links:
        print(link)
    # Menyimpan data ke file CSV
    csv_data = [{'Link': link} for link in links]
    df = pd.DataFrame(csv_data)
    csv_filename = "Scrapy.csv"
    df.to_csv(csv_filename, index=False)
    print("Data telah disimpan ke file CSV:", csv_filename)
    # Menyimpan data ke file Excel (XLS)
    xls_filename = "Scrapy.xls"
    df.to_excel(xls_filename, index=False)
    print("Data telah disimpan ke file Excel (XLS):",
    xls_filename)
# Menjalankan spider menggunakan CrawlerProcess
process = CrawlerProcess()
process.crawl(MySpider)
process.start();
