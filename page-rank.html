from nltk.corpus import stopwords
from nltk.cluster.util import cosine_distance
import numpy as np
import networkx as nx
import PyPDF2 
import sys

Read text dan pembersihan teks

# read teks dan pembersihan teks
def read_article(file_name):

    if file_name.lower().endswith('.txt'):
        f = open(file_name, 'r', encoding="utf8")
        document = f.readlines()
        f.close()

    elif file_name.lower().endswith('.pdf'):
        document = []
        pdfReader = PyPDF2.PdfReader(file_name)
        teks=''
        num_pages = len(pdfReader.pages)
        for i in range(num_pages):
            pageObj = pdfReader.pages[i]
            teks += pageObj.extract_text()
        document.append(teks)

    else:
        sys.exit("File tidak didukung, hanya txt dan pdf yang didukung")
    
    article = document[0].split(". ")
    sentences = []
    for sentence in article:
        sentences.append(sentence.replace("[^a-zA-Z]", " ").replace('\n', '').replace('  ', ' ').split(" "))
    sentences.pop()

    
    return sentences

# ekstrak kalimat yang similar/mirip
def sentence_similarity(sent1, sent2, stopwords=None):
    if stopwords is None:
        stopwords = []
 
    sent1 = [w.lower() for w in sent1]
    sent2 = [w.lower() for w in sent2]
 
    all_words = list(set(sent1 + sent2))
 
    # vector kalimat pertama
    vector1 = [0] * len(all_words)
    for w in sent1:
        if w in stopwords:
            continue
        vector1[all_words.index(w)] += 1
 
    # vector kalimat kedua
    vector2 = [0] * len(all_words)
    for w in sent2:
        if w in stopwords:
            continue
        vector2[all_words.index(w)] += 1
 
    return 1 - cosine_distance(vector1, vector2)
 
def similarity_matrix(sentences, stop_words):
    # similar matrix
    similarity_matrix = np.zeros((len(sentences), len(sentences)))
 
    for idx1 in range(len(sentences)):
        for idx2 in range(len(sentences)):
            if idx1 == idx2: #jika 2 kalimat sama, biarkan
                continue 
            similarity_matrix[idx1][idx2] = sentence_similarity(sentences[idx1], sentences[idx2], stop_words)

    return similarity_matrix

def ringkasanTeks(file_name, top_n=5):
    stop_words = stopwords.words('indonesian')
    summarize_text = []

    # Step 1 - baca text
    sentences =  read_article(file_name)

    # Step 2 - similar matrix dari kalimat
    sentence_similarity_martix = similarity_matrix(sentences, stop_words)

    # Step 3 - RankSentences dari similar matrix
    sentence_similarity_graph = nx.from_numpy_array(sentence_similarity_martix)
    scores = nx.pagerank(sentence_similarity_graph)

    # Step 4 - urutkan RankSentences ASC, ambil kalimat sebanyak top_n dan gabung kalimat.
    ranked_sentence = sorted(((scores[i],s) for i,s in enumerate(sentences)), reverse=True)    
    # print("urutan kalimat berdasarkan RankSentences ", ranked_sentence)
    for i in range(top_n):
      summarize_text.append(" ".join(ranked_sentence[i][1]))

    return ". ".join(summarize_text).replace('  ', ' ').replace(' .', '.')
