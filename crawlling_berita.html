req = requests.get(url)
soup = BeautifulSoup(req.text, 'html.parser')
berita = soup.findAll('div', 'post-list__item')
for item in berita:
  link = item.find('a','d-block')['href']
  judul = item.find('h3','post-list__title').text
  req2 = requests.get(link)
  soup2 = BeautifulSoup(req2.text, 'html.parser')
  try:
    isi = soup2.find('div', 'content').text.replace('\n', '')
  except:
    isi=''
  datas.append([judul, isi])

df = pd.DataFrame(datas, columns=['judul', 'isi'])
